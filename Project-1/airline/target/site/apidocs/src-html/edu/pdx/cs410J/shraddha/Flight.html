001package edu.pdx.cs410J.shraddha;
002
003import edu.pdx.cs410J.AbstractFlight;
004
005
006/**
007 * The class for the airline Project to accept Flight details extended from the Abstract class AbstractFlight
008 */
009public class Flight extends AbstractFlight {
010
011  String fNumber;
012  String srcAirName;
013  String departTime;
014  String destAirName;
015  String arriveTime;
016
017/**
018 * Constructor as the setter function for setting the elements as null for all the elements
019 * */
020  public Flight() {
021    this.fNumber = null;
022    this.srcAirName = null;
023    this.departTime = null;
024    this. destAirName = null;
025    this. arriveTime = null;
026  }
027
028  /**
029   * @return Returns Flight Number - Getter function
030   */
031  @Override
032  public int getNumber() {
033    return Integer.parseInt(fNumber);
034  }
035
036  /**
037   * @return Returns source airline name - Getter function
038   */
039  @Override
040  public String getSource() {
041
042    return this.srcAirName;
043  }
044
045  /**
046   * @return Returns departure time and date - Getter function
047   */
048  @Override
049  public String getDepartureString() {
050
051    return this.departTime;
052  }
053
054  /**
055   * @return Returns destination airline name - Getter function
056   */
057  @Override
058  public String getDestination() {
059
060    return this.destAirName;
061  }
062
063  /**
064   * @return Returns arriaval time and date of flight - Getter function
065   */
066  @Override
067  public String getArrivalString() {
068
069    return this.arriveTime;
070  }
071}
